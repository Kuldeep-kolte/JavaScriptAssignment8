Q.3 Perfect Number Check.

        var Perfect_Check = (number) => {
            var temp = 0;
            for (var i = 1; i <= number / 2; i++) {
                if (number % i === 0) {
                    temp += i;
                }
            }
            if (temp === number && temp !== 0) {
                return "YES";
            }
            else {
                return "NO";
            }
        };

Q.6 You are given a string (STR) of length N, consisting of only the lower case English alphabet.
Your task is to remove all the duplicate occurrences of characters in the string. 

        Input:  abcadeecfb
        
        Output: abcdef
    
        let str = "abcadeecfb";
        document.write("Input : " + str + "
")
        let n = str.length;
        let s = new Set();
        // HashSet doesn't allow repetition of elements
        for (let i = 0; i < n; i++) {
            s.add(str[i]);
        }
        // Print content of the set
        document.write("Output: ")
        for (let a of s) {
            document.write(a);
        }

Q.7 You are given a string (STR) of length N, you have to print the count of all alphabet.(using maps)

            Input:
            abcadeecfb
            
            Output:
            a=2
            b=2
            c=2
            d=1
            e=2
            f=1
    
        let str = "abcadeecfb";
        let map = new Map();
        
        let len = str.length;
        for(let i=0;i< len;i++)
        {

            let char = str.charAt(i);
            if(map.has(char))
            {
                let value = map.get(char);
                map.set(char,value+1);
            }
            else
            {
                map.set(char,1);
            }
        }
